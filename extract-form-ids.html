Method 1: Using Browser Console (Easiest)

  1. Open your Google Form in a new tab
  2. Press F12 or right-click ‚Üí Inspect
  3. Go to the Console tab
  4. Paste and run this code:
// Copy and paste this into the browser console on your Google Form page
(function() {
  const form = document.querySelector('form');
  if (!form) {
    console.error('Form not found! Make sure you are on the Google Form page.');
    return;
  }
  
  const formAction = form.action;
  console.log('%cüìã FORM POST URL:', 'color: #0b3c6d; font-weight: bold; font-size: 14px;');
  console.log(formAction);
  console.log('\n%cüìù FIELD ENTRY IDs:', 'color: #0b3c6d; font-weight: bold; font-size: 14px;');
  
  const inputs = form.querySelectorAll('input, textarea, select');
  const entries = [];
  
  inputs.forEach((input, index) => {
    if (input.name && input.name.startsWith('entry.')) {
      const label = input.closest('[role="listitem"]') || 
                   input.closest('.freebirdFormviewerViewItemsItemItem') ||
                   input.closest('.Qr7Oae');
      const labelText = label ? 
        (label.querySelector('label')?.textContent || 
         label.querySelector('.M7eMe')?.textContent ||
         label.querySelector('[aria-label]')?.getAttribute('aria-label') ||
         'Field ' + (index + 1)) : 'Field ' + (index + 1);
      
      entries.push({
        entryId: input.name,
        label: labelText.trim(),
        type: input.type || input.tagName.toLowerCase()
      });
      
      console.log(`%c${labelText}`, 'color: #28a745; font-weight: bold;');
      console.log(`  Entry ID: ${input.name}`);
      console.log(`  Type: ${input.type || input.tagName.toLowerCase()}\n`);
    }
  });
  
  console.log('\n%cüìã COPY THIS TO YOUR HTML:', 'color: #d32f2f; font-weight: bold; font-size: 14px;');
  console.log(`Form Action: ${formAction}\n`);
  entries.forEach((entry, i) => {
    console.log(`${entry.label}: ${entry.entryId}`);
  });
  
  // Copy to clipboard if possible
  if (navigator.clipboard) {
    const textToCopy = `Form Action: ${formAction}\n\n${entries.map(e => `${e.label}: ${e.entryId}`).join('\n')}`;
    navigator.clipboard.writeText(textToCopy).then(() => {
      console.log('\n%c‚úÖ Copied to clipboard!', 'color: #28a745; font-weight: bold;');
    });
  }
})();

Method 2: View Page Source

  1. Open your Google Form
  2. Right-click on the page ‚Üí View Page Source (or press Ctrl+U)
  3. Press Ctrl+F and search for "entry.
  4. You'll see entries like entry.123456789
  5. Match them with the field labels in the HTML

Method 3: Network Tab

  1. Open your Google Form
  2. Press F12 ‚Üí Go to Network tab
  3. Fill out and submit the form
  4. Look for a request to formResponse
  5. Check the Payload tab to see all entry IDs
